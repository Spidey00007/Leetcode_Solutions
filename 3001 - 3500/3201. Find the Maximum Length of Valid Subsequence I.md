class Solution {
public:
    int solve(bool check, vector<int>& nums) {
        int count=0;
        for(int& num:nums) {
            if(num%2 == check) {
                count++;
                check=!check;
            }
        }
        return count;
    }

    int maximumLength(vector<int>& nums) {
        int ans=2;
        ans=max({ans, solve(true, nums), solve(false, nums)});

        int odd=0, even=0;
        for(int& num:nums) {
            (num%2==0) ? even++ : odd++;
        }

        ans=max({ans, even, odd});

        return ans;
    }
};
