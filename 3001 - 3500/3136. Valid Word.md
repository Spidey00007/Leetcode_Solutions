class Solution {
public:
    bool isValid(string word) {
        if(word.size()<3) {
            return false;
        }

        bool num=false, vowel=false, consonant=false, other=false;
        for(char& ch:word) {
            if(!isalnum(ch)) {
                return false;
            }

            if(isdigit(ch)) {
                num=true;
                continue;
            }

            ch=tolower(ch);

            if(ch=='a' || ch=='e' || ch=='i' || ch=='o' || ch=='u') {
                vowel=true;
            } else {
                consonant=true;
            }
        }

        if(!vowel || !consonant) {
            return false;
        }
        return true;
    }
};
