class Solution {
public:
    bool isCycleDFS(int u, vector<bool>& visited, vector<bool>& inRecursion, vector<vector<int>>& graph) {
        visited[u]=true;
        inRecursion[u]=true;

        for(int& v:graph[u]) {
            if(!visited[v] && isCycleDFS(v, visited, inRecursion, graph)) {
                return true;
            } else if(inRecursion[v]) {
                return true;
            }
        }

        inRecursion[u]=false;
        return false;
    }

    vector<int> eventualSafeNodes(vector<vector<int>>& graph) {
        int n=graph.size();
        vector<bool> visited(n, false), inRecursion(n, false);

        for(int i=0; i<n; i++) {
            if(!visited[i]) {
                isCycleDFS(i, visited, inRecursion, graph);
            }
        }

        vector<int> result;
        for(int i=0; i<n; i++) {
            if(!inRecursion[i]) {
                result.push_back(i);
            }
        }
        return result;
    }
};
