# üåì Is the Graph Bipartite? ‚Äì BFS & DFS Explorations üß≠

### üü¢ BFS Approach

#### ‚è≥ Time Complexity: O(V + E)

#### üì¶ Space Complexity: O(V)

```cpp
class Solution {
public:
    bool BFS(int u, vector<vector<int>>& graph, vector<int>& color) {
        queue<int> q;
        q.push(u);

        while(!q.empty()) {
            int u=q.front();
            q.pop();

            for(int& v:graph[u]) {
                if(color[u] == color[v]) {
                    return false;
                } else if(color[v] == -1) {
                    color[v]=1-color[u];
                    q.push(v);
                }
            }
        }
        return true;
    }

    bool isBipartite(vector<vector<int>>& graph) {
        int n=graph.size();
        vector<int> color(n, -1);

        for(int i=0; i<n; i++) {
            if(color[i] == -1) {
                color[i]=0;
                if(!BFS(i, graph, color)) {
                    return false;
                }
            }
        }
        return true;
    }
};
```

### üîµ DFS Approach (Style 1)

#### ‚è≥ Time Complexity: O(V + E)

#### üì¶ Space Complexity: O(V)

class Solution {
public:
    bool DFS(int u, vector<vector<int>>& graph, vector<int>& color) {
        for(int& v:graph[u]) {
            if(color[v] == -1) {
                color[v]=1-color[u];
                if(!DFS(v, graph, color)) {
                    return false;
                }
            } else if(color[u] == color[v]) {
                return false;
            }
        }
        return true;
    }

    bool isBipartite(vector<vector<int>>& graph) {
        int n=graph.size();

                color[i]=0;
                if(!DFS(i, graph, color)) {
                    return false;
                }
            }
        }
        return true;
    }
};

### üî¥ DFS Approach (Style 2 ‚Äì Alternate Condition Ordering)

#### ‚è≥ Time Complexity: O(V + E)

#### üì¶ Space Complexity: O(V)

```cpp
class Solution {
public:
    bool DFS(int u, vector<vector<int>>& graph, vector<int>& color) {
        for(int& v:graph[u]) {
            if(color[u] == color[v]) {
                return false;
            }
            if(color[v]!=-1) {
                continue;
            }
            color[v]=1-color[u];

            if(!DFS(v, graph, color)) {
                return false;
            }
        }
        return true;
    }

    bool isBipartite(vector<vector<int>>& graph) {
        int n=graph.size();
        vector<int> color(n, -1);

        for(int i=0; i<n; i++) {
            if(color[i] == -1) {
                color[i]=0;
                if(!DFS(i, graph, color)) {
                    return false;
                }
            }
        }
        return true;
    }
};
```
