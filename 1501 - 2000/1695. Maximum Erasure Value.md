class Solution {
public:
    int maximumUniqueSubarray(vector<int>& nums) {
        unordered_set<int> st;
        int l=0, ans=0, sum=0;

        for(int r=0; r<nums.size(); r++) {
            while(st.find(nums[r]) != st.end()) {
                sum-=nums[l];
                st.erase(nums[l]);
                l++;
            }
            st.insert(nums[r]);
            sum+=nums[r];
            ans=max(ans, sum);
        }
        return ans;
    }
};
