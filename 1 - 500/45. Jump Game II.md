> **Time Complexity:** O(n)  
> **Space Complexity:** O(1)

```cpp
class Solution {
public:
    int jump(vector<int>& nums) {
        int n=nums.size(), l=0, r=0, step=0;
        while(r < n-1) {
            int maxi=r;
            for(int i=l; i<=r; i++) {
                maxi=max(maxi, i+nums[i]);
            }

            step++;
            if(maxi >= n-1) {
                break;
            }
            l=r+1;
            r=maxi;
        }
        return step;
    }
};
```

## ðŸŸ¡ Bottom-Up Dynamic Programming
> **Time Complexity:** O(n*k)  
> **Space Complexity:** O(n)

```cpp
class Solution {
public:
    int jump(vector<int>& nums) {
        int n=nums.size();
        vector<int> dp(n, n);
        dp[0]=0;

        for(int i=0; i<nums.size(); i++) {
            for(int step=1; step<=nums[i]; step++) {
                if(i+step < n) {
                    dp[i+step]=min(dp[i+step], dp[i]+1);
                }
            }
        }
        return dp[n-1];
    }
};
```

## ðŸ” Top-Down Dynamic Programming (Recursion + Memoization)
> **Time Complexity:** O(n*k)  
> **Space Complexity:** O(n) + O(n) recursion stack

```cpp
class Solution {
public:
    vector<int> dp;
    int solve(int i, vector<int>& nums) {
        if(i >= nums.size()-1) {
            return 0;
        }
        if(dp[i] != -1) {
            return dp[i];
        }

        int ans=INT_MAX;
        for(int step=1; step<=nums[i]; step++) {
            int temp=solve(i+step, nums);
            if(temp != INT_MAX) {
                ans=min(ans, 1+solve(i+step, nums));
            }
        }
        return dp[i]=ans;
    }

    int jump(vector<int>& nums) {
        dp.resize(nums.size(), -1);
        return solve(0, nums);
    }
};
```
