class Solution {
public:
    int ladderLength(string beginWord, string endWord, vector<string>& wordList) {
        unordered_map<string, bool> visited;
        for(string& str:wordList) {
            visited[str]=false;
        }

        int ans=0;
        queue<string> q;
        q.push(beginWord);

        while(!q.empty()) {
            int n=q.size();
            ans++;

            while(n--) {
                string str=q.front();
                if(str==endWord) {
                    return ans;
                }
                visited[str]=true;
                q.pop();

                for(int i=0; i<str.size(); i++) {
                    for(int j=0; j<26; j++) {
                        string temp=str;
                        temp[i]='a'+j;

                        if(visited.find(temp)!=visited.end() && visited[temp]==false) {
                            q.push(temp);
                        }
                    }
                }
            }
        }
        return 0;
    }
};
