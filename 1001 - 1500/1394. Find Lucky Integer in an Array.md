class Solution {
public:
    int findLucky(vector<int>& arr) {
        unordered_map<int, int> mp;
        for(int& num:arr) {
            mp[num]++;
        }

        int ans=-1;
        for(auto [num, freq] : mp) {
            if(num==freq) ans=max(ans, num);
        }
        return ans;
    }
};
